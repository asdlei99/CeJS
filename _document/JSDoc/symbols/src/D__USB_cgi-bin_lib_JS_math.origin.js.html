149 function floorSurt(n){
150  if(isNaN(n))return;
151  var g=0,v,h,t;
152  while((t=2*g)<(v=n-g*g)){
153   h=1;//alert(t+','+v);
154   while(h*(h+t)<=v)h<<=1;
155   g+=h>>1;
156  }
157  //alert('end:'+t+','+v);
158  return g;
159 }
