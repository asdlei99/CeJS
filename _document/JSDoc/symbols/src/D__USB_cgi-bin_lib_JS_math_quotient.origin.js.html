143 function quotient(numerator,denominator,base,approximate){
144  if(typeof numerator=='object'&&numerator instanceof quotient)return numerator;//numerator.type=='quotient'
145  if(isNaN(numerator))numerator=0;
146  if(!denominator||isNaN(denominator))denominator=1;
147  else if(denominator<0)denominator=-denominator,numerator=-numerator;
148 
149  var q=qNum(numerator);	//	qNum需test，並回傳(分子,分母,誤差)！
150  if(!q)numerator=0;
151  else if(approximate||!q[2])numerator=q[0],denominator*=q[1];	//	無誤差時使用之
152  else while(numerator%1||denominator%1)numerator*=10,denominator*=10;	//	化為整數
153 
154  //	value
155  this.n=numerator,this.d=denominator;//this.type='quotient';
156  return this;
157 }
